 3(Vcc) - 2
 7(RS) - 5
 9(WR) - 3
11(RD) - 1
13(DB8) - 29
15(DB9) - 31
17(DB10) - 12
18(T_Clk) - 23
19(DB11) - 16
20(T_CS) - 26
21(DB12) - 18
22(T_Din) - 19
23(DB13) - 22
25(DB14) - 24
26(T_DO) - 21
27(DB15) - 32
28(T_IRQ) - 36
29(CS) - 13
33(REST) - 11
37(LED) - Vcc
